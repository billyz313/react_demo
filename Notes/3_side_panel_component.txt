export class SideSlider extends React.Component {
  showSettings(event) {

  }

  render() {
    return (
      <Menu>
        <a id="home" className="menu-item" href="/">

        <a id="about" className="menu-item" href="/about">

        </a>
        <a onClick={this.showSettings} className="menu-item--small" href="">
          <span>Settings</span>
        </a>
      </Menu>
    );
  }
}
